# 常用运算符
## != 与 <> （不等于）
&nbsp;&nbsp; != 和 <> 是一样的，都是不包含NULL的情况，包含NULL 需要使用IS 
```sql
  # 测试案例: 5.7.34-debug
  create table operation_test (id bigint not null auto_increment primary key , name varchar(50)) character set utf8mb4 collate utf8mb4_unicode_ci;

  insert into operation_test(name) values('张三'),('李四'),('王五'),(''),(NULL);

  mysql> select * from operation_test;
  +----+--------+
  | id | name   |
  +----+--------+
  |  1 | 张三   |
  |  2 | 李四   |
  |  3 | 王五   |
  |  4 |        |
  |  5 | NULL   |
  +----+--------+
  5 rows in set (0.00 sec)

  mysql> select * from operation_test where name != '张三';
  +----+--------+
  | id | name   |
  +----+--------+
  |  2 | 李四   |
  |  3 | 王五   |
  |  4 |        |
  +----+--------+
  3 rows in set (0.00 sec)

  mysql> select * from operation_test where name <> '张三';
  +----+--------+
  | id | name   |
  +----+--------+
  |  2 | 李四   |
  |  3 | 王五   |
  |  4 |        |
  +----+--------+
  3 rows in set (0.00 sec)

  mysql> select * from operation_test where name <> '张三' or name IS NULL;
  +----+--------+
  | id | name   |
  +----+--------+
  |  2 | 李四   |
  |  3 | 王五   |
  |  4 |        |
  |  5 | NULL   |
  +----+--------+
  4 rows in set (0.00 sec)
 
```

## = 、<=> 、 IS  （等于）
&nbsp;&nbsp;IS 专门用于判断是否为NULL，而=是用来判断非NULL以外的所有数据类型，<=>是前两者合起来(判断等于，且可以用于NULL判断)。
```sql
   # 测试案例: 5.7.34-debug
  create table operation_test (id bigint not null auto_increment primary key , name varchar(50)) character set utf8mb4 collate utf8mb4_unicode_ci;

  insert into operation_test(name) values('张三'),('李四'),('王五'),(''),(NULL);

  mysql> select * from operation_test;
  +----+--------+
  | id | name   |
  +----+--------+
  |  1 | 张三   |
  |  2 | 李四   |
  |  3 | 王五   |
  |  4 |        |
  |  5 | NULL   |
  +----+--------+
  5 rows in set (0.00 sec)

  mysql> select * from operation_test where name = NULL;
  Empty set (0.01 sec)

  mysql> select * from operation_test where name IS  NULL;
  +----+------+
  | id | name |
  +----+------+
  |  5 | NULL |
  +----+------+
  1 row in set (0.00 sec)
  
  mysql> select * from operation_test where name <=>  NULL;
  +----+------+
  | id | name |
  +----+------+
  |  5 | NULL |
  +----+------+
  1 row in set (0.01 sec)
  
  mysql> select * from operation_test where name <=>  '张三';
  +----+--------+
  | id | name   |
  +----+--------+
  |  1 | 张三   |
  +----+--------+
  1 row in set (0.01 sec)

```
